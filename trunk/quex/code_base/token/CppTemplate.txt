#ifndef __QUEX_INCLUDE_GUARD__TOKEN__GENERATED__QUEX_TYPE_DOKEN_WITHOUT_NAMESPACE
#define __QUEX_INCLUDE_GUARD__TOKEN__GENERATED__QUEX_TYPE_DOKEN_WITHOUT_NAMESPACE

#include <quex/code_base/definitions>
 
QUEX_NAMESPACE_TOKEN_OPEN

class QUEX_TYPE_DOKEN_WITHOUT_NAMESPACE {
public:
    QUEX_TYPE_DOKEN_WITHOUT_NAMESPACE();
    QUEX_TYPE_DOKEN_WITHOUT_NAMESPACE(const QUEX_TYPE_DOKEN_WITHOUT_NAMESPACE& That);
    void __copy(const QUEX_TYPE_DOKEN_WITHOUT_NAMESPACE& That);
    /* operator=(..): USE WITH CAUTION--POSSIBLE MAJOR PERFORMANCE DECREASE! 
     *                BETTER USE __copy(That)                                */
    QUEX_TYPE_DOKEN_WITHOUT_NAMESPACE operator=(const QUEX_TYPE_DOKEN_WITHOUT_NAMESPACE& That) 
    { __copy(That); return *this; }
    $$VIRTUAL_DESTRUCTOR$$~QUEX_TYPE_DOKEN_WITHOUT_NAMESPACE();

$$DISTINCT_MEMBERS$$
$$UNION_MEMBERS$$
public:
$$SETTERS_GETTERS$$
    void set(const QUEX_TYPE_TOKEN_WITH_NAMESPACE_ID ID) { _id = ID; }
$$QUICK_SETTERS$$

    private:
        QUEX_TYPE_TOKEN_WITH_NAMESPACE_ID    _id;
    public:
        QUEX_TYPE_TOKEN_WITH_NAMESPACE_ID    type_id() const      { return _id; }
        static const char*    map_id_to_name(QUEX_TYPE_TOKEN_WITH_NAMESPACE_ID);
        const std::string     type_id_name() const { return map_id_to_name(_id); }

#   ifdef     QUEX_OPTION_TOKEN_STAMPING_WITH_LINE_AND_COLUMN
#       ifdef QUEX_OPTION_LINE_NUMBER_COUNTING
    private:
        QUEX_TYPE_TOKEN_WITH_NAMESPACE_LINE_N  _line_n;
    public:
        QUEX_TYPE_TOKEN_WITH_NAMESPACE_LINE_N    line_number() const                           { return _line_n; }
        void                set_line_number(const QUEX_TYPE_TOKEN_WITH_NAMESPACE_LINE_N Value) { _line_n = Value; }
#       endif
#       ifdef  QUEX_OPTION_COLUMN_NUMBER_COUNTING
    private:
        QUEX_TYPE_TOKEN_WITH_NAMESPACE_COLUMN_N  _column_n;
    public:
        QUEX_TYPE_TOKEN_WITH_NAMESPACE_COLUMN_N  column_number() const                             { return _column_n; }
        void                      set_column_number(const QUEX_TYPE_TOKEN_WITH_NAMESPACE_COLUMN_N Value) { _column_n = Value; }
#       endif
#   endif
    public:
$$BODY$$
};

inline
QUEX_TYPE_DOKEN_WITHOUT_NAMESPACE::QUEX_TYPE_DOKEN_WITHOUT_NAMESPACE()
{
$$CONSTRUCTOR$$
}

inline
QUEX_TYPE_DOKEN_WITHOUT_NAMESPACE::QUEX_TYPE_DOKEN_WITHOUT_NAMESPACE(const QUEX_TYPE_DOKEN_WITHOUT_NAMESPACE& That)
{
    __copy(That);
$$CONSTRUCTOR$$
}

inline
QUEX_TYPE_DOKEN_WITHOUT_NAMESPACE::~QUEX_TYPE_DOKEN_WITHOUT_NAMESPACE()
{
$$DESTRUCTOR$$
}

inline void
QUEX_TYPE_DOKEN_WITHOUT_NAMESPACE::__copy(const QUEX_TYPE_DOKEN_WITHOUT_NAMESPACE& That)
{
$$COPY$$
    /* If the user even misses to copy the token id, then there's
     * something seriously wrong.                                 */
    __quex_assert(this->_id == That._id);
#ifdef     QUEX_OPTION_TOKEN_STAMPING_WITH_LINE_AND_COLUMN
#   ifdef QUEX_OPTION_LINE_NUMBER_COUNTING
    __quex_assert(_line_n == That._line_n);
#   endif
#   ifdef  QUEX_OPTION_COLUMN_NUMBER_COUNTING
    __quex_assert(_column_n == That._column_n);
#   endif
#endif
}

QUEX_NAMESPACE_TOKEN_CLOSE

#endif /* __QUEX_INCLUDE_GUARD__TOKEN__GENERATED__QUEX_TYPE_DOKEN_WITHOUT_NAMESPACE */
